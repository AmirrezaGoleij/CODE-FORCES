# CODE-FORCES


#include <iostream>

using namespace std;

int main()
{
    int n, a, X(0), Y(0), Z(-1);

    cin >> n;

    while (n--)

    {
        cin >> a;

        if (a == 1)

        {
            X += 1;

    if (Y > 0)

            {

                Y -= 1;
            }
        }

        else

        {
            Y += 1;

            if (Y > Z)

            {

                Z = Y;
            }
        }
    }

    cout << X + Z << endl;

    return 0;
}
